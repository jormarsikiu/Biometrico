43 int read_jpeg_file( string filename )
44 {
45  /* these are standard libjpeg structures for reading(decompression) */
46  struct jpeg_decompress_struct cinfo;
47  struct jpeg_error_mgr jerr;
48  /* libjpeg data structure for storing one row, that is, scanline of an image */
49  JSAMPROW row_pointer[1];
50 
51  FILE *infile = fopen( filename.c_str(), "rb" );
52  unsigned long location = 0;
53  size_t i = 0;
54 
55  if ( !infile )
56  {
57  printf("Error opening jpeg file %s\n!", filename.c_str() );
58  return -1;
59  }
60  /* here we set up the standard libjpeg error handler */
61  cinfo.err = jpeg_std_error( &jerr );
62  /* setup decompression process and source, then read JPEG header */
63  jpeg_create_decompress( &cinfo );
64  /* this makes the library read from infile */
65  jpeg_stdio_src( &cinfo, infile );
66  /* reading the image header which contains image information */
67  jpeg_read_header( &cinfo, TRUE );
68  /* Uncomment the following to output image information, if needed. */
69  /*--
70  printf( "JPEG File Information: \n" );
71  printf( "Image width and height: %d pixels and %d pixels.\n", cinfo.image_width, cinfo.image_height );
72  printf( "Color components per pixel: %d.\n", cinfo.num_components );
73  printf( "Color space: %d.\n", cinfo.jpeg_color_space );
74  --*/
75  /* Start decompression jpeg here */
76  jpeg_start_decompress( &cinfo );
77 
78  /* allocate memory to hold the uncompressed image */
79  raw_image = (unsigned char*)malloc( cinfo.output_width*cinfo.output_height*cinfo.num_components );
80  /* now actually read the jpeg into the raw buffer */
81  row_pointer[0] = (unsigned char *)malloc( cinfo.output_width*cinfo.num_components );
82  /* read one scan line at a time */
83  while( cinfo.output_scanline < cinfo.image_height )
84  {
85  jpeg_read_scanlines( &cinfo, row_pointer, 1 );
86  for( i=0; i<cinfo.image_width*cinfo.num_components;i++)
87  raw_image[location++] = row_pointer[0][i];
88  }
89  /* wrap up decompression, destroy objects, free pointers and close open files */
90  jpeg_finish_decompress( &cinfo );
91  jpeg_destroy_decompress( &cinfo );
92  free( row_pointer[0] );
93  fclose( infile );
94  /* yup, we succeeded! */
95  return 1;
96 }

116 
117  if ( !outfile )
118  {
119  printf("Error opening output jpeg file %s\n!", filename.c_str() );
120  return -1;
121  }
122  cinfo.err = jpeg_std_error( &jerr );
123  jpeg_create_compress(&cinfo);
124  jpeg_stdio_dest(&cinfo, outfile);
125 
126  /* Setting the parameters of the output file here */
127  cinfo.image_width = width;
128  cinfo.image_height = height;
129  cinfo.input_components = bytes_per_pixel;
130  cinfo.in_color_space = color_space;
131  /* default compression parameters, we shouldn't be worried about these */
132  jpeg_set_defaults( &cinfo );
133  /* Now do the compression .. */
134  jpeg_start_compress( &cinfo, TRUE );
135  /* like reading a file, this time write one row at a time */
136  while( cinfo.next_scanline < cinfo.image_height )
137  {
138  row_pointer[0] = &raw_image[ cinfo.next_scanline * cinfo.image_width * cinfo.input_components];
139  jpeg_write_scanlines( &cinfo, row_pointer, 1 );
140  }
141  /* similar to read file, clean up after we're done compressing */
142  jpeg_finish_compress( &cinfo );
143  jpeg_destroy_compress( &cinfo );
144  fclose( outfile );
145  /* success code is 1! */
146  return 1;

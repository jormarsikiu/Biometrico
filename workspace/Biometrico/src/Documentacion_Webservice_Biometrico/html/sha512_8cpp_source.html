55 void SHA512::transform(const unsigned char *message, unsigned int block_nb)
56 {
57  uint64 w[80];
58  uint64 wv[8];
59  uint64 t1, t2;
60  const unsigned char *sub_block;
61  int i, j;
62  for (i = 0; i < (int) block_nb; i++) {
63  sub_block = message + (i << 7);
64  for (j = 0; j < 16; j++) {
65  SHA2_PACK64(&sub_block[j << 3], &w[j]);
66  }
67  for (j = 16; j < 80; j++) {
68  w[j] = SHA512_F4(w[j - 2]) + w[j - 7] + SHA512_F3(w[j - 15]) + w[j - 16];
69  }
70  for (j = 0; j < 8; j++) {
71  wv[j] = m_h[j];
72  }
73  for (j = 0; j < 80; j++) {
74  t1 = wv[7] + SHA512_F2(wv[4]) + SHA2_CH(wv[4], wv[5], wv[6])
75  + sha512_k[j] + w[j];
76  t2 = SHA512_F1(wv[0]) + SHA2_MAJ(wv[0], wv[1], wv[2]);
77  wv[7] = wv[6];
78  wv[6] = wv[5];
79  wv[5] = wv[4];
80  wv[4] = wv[3] + t1;
81  wv[3] = wv[2];
82  wv[2] = wv[1];
83  wv[1] = wv[0];
84  wv[0] = t1 + t2;
85  }

107 void SHA512::update(const unsigned char *message, unsigned int len)
108 {
109  unsigned int block_nb;
110  unsigned int new_len, rem_len, tmp_len;
111  const unsigned char *shifted_message;
112  tmp_len = SHA384_512_BLOCK_SIZE - m_len;
113  rem_len = len < tmp_len ? len : tmp_len;
114  memcpy(&m_block[m_len], message, rem_len);
115  if (m_len + len < SHA384_512_BLOCK_SIZE) {
116  m_len += len;
117  return;
118  }
119  new_len = len - rem_len;
120  block_nb = new_len / SHA384_512_BLOCK_SIZE;
121  shifted_message = message + rem_len;
122  transform(m_block, 1);
123  transform(shifted_message, block_nb);
124  rem_len = new_len % SHA384_512_BLOCK_SIZE;
125  memcpy(m_block, &shifted_message[block_nb << 7], rem_len);
126  m_len = rem_len;
127  m_tot_len += (block_nb + 1) << 7;
128 }

131 {
132  unsigned int block_nb;
133  unsigned int pm_len;
134  unsigned int len_b;
135  int i;
136  block_nb = 1 + ((SHA384_512_BLOCK_SIZE - 17)
137  < (m_len % SHA384_512_BLOCK_SIZE));
138  len_b = (m_tot_len + m_len) << 3;
139  pm_len = block_nb << 7;
140  memset(m_block + m_len, 0, pm_len - m_len);
141  m_block[m_len] = 0x80;
142  SHA2_UNPACK32(len_b, m_block + pm_len - 4);
143  transform(m_block, block_nb);
144  for (i = 0 ; i < 8; i++) {
145  SHA2_UNPACK64(m_h[i], &digest[i << 3]);
146  }
